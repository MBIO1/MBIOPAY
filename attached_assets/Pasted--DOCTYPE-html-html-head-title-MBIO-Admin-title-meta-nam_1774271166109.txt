<body>

<h2>MBIO SYSTEM DASHBOARD</h2>

<button onclick="loadAll()">Refresh</button>

<div id="overview"></div>
<div id="users"></div>
<div id="orders"></div>
<div id="withdrawals"></div>

<script>

async function loadAll() {
  loadOverview();
  loadUsers();
  loadOrders();
  loadWithdrawals();
}

async function loadOverview() {
  const res = await fetch("/admin/overview");
  const d = await res.json();

  document.getElementById("overview").innerHTML = `
    <div class="card">
      <h3>System</h3>
      Users: ${d.users}<br/>
      Balance: ${d.systemBalance}<br/>
      Hot Wallet: ${d.hotWallet}<br/>
      Pending Withdrawals: ${d.pendingWithdrawals}
    </div>
  `;
}

async function loadUsers() {
  const res = await fetch("/admin/users");
  const users = await res.json();

  document.getElementById("users").innerHTML = `
    <div class="card">
      <h3>Users</h3>
      ${users.map(u => `
        <div>
          ID:${u.id} | Balance:${u.balance} | Risk:${u.riskScore}
          <button onclick="freeze(${u.id})">Freeze</button>
          <button onclick="reset(${u.id})">Reset</button>
        </div>
      `).join("")}
    </div>
  `;
}

async function loadOrders() {
  const res = await fetch("/admin/orders");
  const data = await res.json();

  document.getElementById("orders").innerHTML = `
    <div class="card">
      <h3>Orders</h3>
      ${data.map(o => `
        <div>#${o.id} | ${o.amount} | ${o.status}</div>
      `).join("")}
    </div>
  `;
}

async function loadWithdrawals() {
  const res = await fetch("/admin/withdrawals");
  const data = await res.json();

  document.getElementById("withdrawals").innerHTML = `
    <div class="card">
      <h3>Withdrawals</h3>
      ${data.map(w => `
        <div>#${w.id} | ${w.amount} | ${w.status}</div>
      `).join("")}
    </div>
  `;
}

async function freeze(id) {
  await fetch("/admin/freeze/" + id, { method:"POST" });
  loadAll();
}

async function reset(id) {
  await fetch("/admin/reset-risk/" + id, { method:"POST" });
  loadAll();
}

</script>

</body>
</html>
