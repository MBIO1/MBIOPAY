/* 🔥 REALISTIC ACTIVITY FEED */
const names = ["John", "Sarah", "David", "Michael", "Grace", "Daniel"];
const networks = ["MTN", "Airtel"];

function randomItem(arr){
  return arr[Math.floor(Math.random() * arr.length)];
}

function generateActivity(){
  const name = randomItem(names);
  const amount = Math.floor(Math.random() * 200) + 20;
  const network = randomItem(networks);

  return `${name} sent ${amount} USD to ${network}`;
}

function renderFeed(){
  const feed = document.getElementById("feed");
  feed.innerHTML = "";

  for(let i=0;i<5;i++){
    const div = document.createElement("div");
    div.className = "feed-item";
    div.innerText = generateActivity() + " • just now";
    feed.appendChild(div);
  }
}

renderFeed();

/* refresh every 10s */
setInterval(renderFeed, 10000);

// --- ACTIVITY FEED (SAFE + SANITIZED) ---
app.get("/api/activity", (req, res) => {
  db.all(
    `SELECT phone, network, amount, status, created_at 
     FROM orders 
     WHERE status IN ('processing','completed')
     ORDER BY id DESC 
     LIMIT 10`,
    [],
    (err, rows) => {
      if (err) return res.json([]);

      const sanitized = rows.map(r => {
        // mask phone (privacy)
        const masked = r.phone
          ? r.phone.slice(0,3) + "****" + r.phone.slice(-2)
          : "User";

        return {
          user: masked,
          network: r.network,
          amount: Math.round(r.amount || 0),
          time: "just now" // keep simple (no timestamps needed publicly)
        };
      });

      res.json(sanitized);
    }
  );
});

db.run(`
CREATE TABLE IF NOT EXISTS orders (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  phone TEXT,
  network TEXT,
  amount REAL,
  status TEXT,
  txid TEXT,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP
)
`);

// --- REAL ACTIVITY FEED ---
async function loadFeed() {
  try {
    const res = await fetch("/api/activity");
    const data = await res.json();

    const feed = document.getElementById("feed");
    feed.innerHTML = "";

    if (!data.length) {
      feed.innerHTML = "<p class='muted'>No recent activity yet</p>";
      return;
    }

    data.forEach(item => {
      const div = document.createElement("div");
      div.className = "feed-item";

      div.innerText = 
        `${item.user} sent ${item.amount} to ${item.network} • ${item.time}`;

      feed.appendChild(div);
    });

  } catch (err) {
    document.getElementById("feed").innerHTML =
      "<p class='muted'>Activity unavailable</p>";
  }
}

// load + refresh
loadFeed();
setInterval(loadFeed, 10000);
</script>
