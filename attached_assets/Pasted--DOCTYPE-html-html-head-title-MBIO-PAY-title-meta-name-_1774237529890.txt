<script>
let orderId = null;
let userAmount = 0;

// STEP 1 → GET QUOTE
async function getQuote() {
  const amount = document.getElementById("amount").value;
  userAmount = amount;

  const res = await fetch("/quote?amount=" + amount);
  const data = await res.json();

  document.getElementById("ugx").innerText =
    data.payoutUGX.toFixed(0) + " UGX";

  document.getElementById("step1").classList.add("hidden");
  document.getElementById("step2").classList.remove("hidden");
}

// STEP 2 → CREATE ORDER
async function createOrder() {
  const phone = document.getElementById("phone").value;
  const network = document.getElementById("network").value;

  const res = await fetch("/create-order", {
    method: "POST",
    headers: {"Content-Type":"application/json"},
    body: JSON.stringify({ phone, network })
  });

  const data = await res.json();
  orderId = data.orderId;

  document.getElementById("wallet").innerText = data.address;
  document.getElementById("orderText").innerText =
    "Order ID: " + orderId;

  document.getElementById("step2").classList.add("hidden");
  document.getElementById("step3").classList.remove("hidden");
}

// STEP 3 → CHECK STATUS
async function checkStatus() {
  const res = await fetch("/order/" + orderId);
  const data = await res.json();

  document.getElementById("status").innerText =
    "Status: " + data.status;
}
</script>
